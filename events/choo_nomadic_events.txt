choo_nomadic.1 = {
	type = country_event
	hidden = yes
	
	trigger = {
	government = nomadic_tribe
	any_character = {
	OR = { is_general = yes is_ruler = yes }  
	NOT = { is_target_in_variable_list = { name = nomadic_leaders_list target = this } }
	}

	}

	immediate = {
	every_character = {
			limit = {
			OR = { is_general = yes is_ruler = yes }  
			NOT = {
			is_target_in_variable_list = { name = nomadic_leaders_list target = this }
			}
			}
			save_temporary_scope_as = to_add
			root = {
				add_to_variable_list = {
					name = nomadic_leaders_list
					target = scope:to_add
				}
			}
		}

	
	}
}
