ba_add_slaves_button = {
	scope = province

	is_valid = {
		always = yes
	}
	effect = {
		create_pop = slaves
	}
}
ba_kill_slaves_button = {
	scope = province

	is_valid = {
		always = yes
	}
	effect = {
		random_pops_in_province = {
			limit = {
				pop_type = slaves
			}
			kill_pop = yes
		}
	}
}
